```python
# ┌──────────────────────────────────────────────────────────────────────────┐
# │  TOXIQ  —  Drug Toxicity Prediction Engine                                │
# │  Stack: Python · RDKit · Morgan Fingerprints · XGBoost                   │
# └──────────────────────────────────────────────────────────────────────────┘

from rdkit      import Chem
from rdkit.Chem import AllChem
from xgboost    import XGBClassifier

class ToxIQ:
    """
    Multi-target toxicity screening for drug discovery pipelines.
    Converts SMILES → Morgan Fingerprints → XGBoost inference.
    """
    def predict(self, smiles: str) -> dict:
        mol         = Chem.MolFromSmiles(smiles)
        fingerprint = AllChem.GetMorganFingerprintAsBitVect(mol, 2, 2048)
        result      = self.model.predict_proba([fingerprint])
        return {
            "compound"   : smiles,
            "toxic"      : bool(result[0][1] > 0.5),
            "confidence" : f"{result[0][1]*100:.1f}%"
            # ✓ SMILES  ✓ Morgan FP  ✓ XGBoost  ✓ Drug Discovery
        }
```
